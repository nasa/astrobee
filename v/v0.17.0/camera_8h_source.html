48 class CameraNodelet : public ff_util::FreeFlyerNodelet {
49  public:
50  // The size of the image message ring buffer for publishing
51  // grayscale images. 30 images = 35.2 MB of buffer space. This
52  // number is so large because it sets the lifetime for each image
53  // message sent out. Eventually we'll write over the pointer we've
54  // handed out. At 15 Hz, 30 frames means an image will stay valid
55  // for 2.0 seconds. Localization can process at 1 Hz, so it only
56  // needs an image for 1.0 second.
57  static constexpr size_t kImageMsgBuffer = 30;
58 
59  // The size of the image message ring buffer for publishing raw
60  // Bayer format images. The same comments apply about message
61  // lifetime, but in this case we're publishing at a lower rate and
62  // expecting the subscriber callback to at most de-Bayer the image
63  // before passing it on, so the buffer doesn't need to be so long.
64  static constexpr size_t kBayerImageMsgBufferLength = 5;
