151 void EigenCovarianceToMsg(const Eigen::Matrix<double, 6, 6>& covariance, ArrayType& covariance_array) {
152  for (int i = 0; i < 6; ++i) {
153  for (int j = 0; j < 6; ++j) {
154  covariance_array[i*6 + j] = covariance(i, j);
