63  boost::shared_ptr<GaussianFactor> linearize(const Values& values) const override {
64  typename Base::Cameras cameras = this->cameras(values);
65  // if (!this->triangulateForLinearize(cameras)) return boost::make_shared<JacobianFactorSVD<Dim, 2>>(this->keys());
66  const auto result = this->triangulateSafe(cameras);
67  // Adapted from SmartFactorBase::CreateJacobianSVDFactor
68  size_t m = this->keys().size();
69  typename Base::FBlocks F;
70  Vector b;
71  const size_t M = ZDim * m;
72  Matrix E0(M, M - 3);
73 
74  // Handle behind camera result with rotation only factors (see paper)
75  // Degenerate result tends to lead to solve failures, so return empty factor in this case
76  if (result.valid()) {
77  this->computeJacobiansSVD(F, E0, b, cameras, *(this->point()));
78  } else if (useForRotationOnly(result)) { // Rotation only factor
79  Unit3 backProjected = cameras[0].backprojectPointAtInfinity(this->measured().at(0));
80  // Cheirality error can still occur with backprojection
81  try {
82  this->computeJacobiansSVD(F, E0, b, cameras, backProjected);
83  } catch (...) {
84  return boost::make_shared<JacobianFactorSVD<Dim, 2>>(this->keys());
85  }
86  } else { // Empty factor // NOLINT
87  return boost::make_shared<JacobianFactorSVD<Dim, 2>>(this->keys());
88  }
89  return createRegularJacobianFactorSVD<Dim, ZDim>(this->keys(), F, E0, b);
90  }

97  double error(const Values& values) const override {
98  if (this->active(values)) {
99  try {
100  const double total_reprojection_loss = this->totalReprojectionError(this->cameras(values));
101  const auto result = this->point();
102  if (!result.valid() && !useForRotationOnly(result)) return 0.0;
103  // Multiply by 2 since totalReporjectionError divides mahal distance by 2, and robust_model_->loss
104  // expects mahal distance
105  const double loss = robust_ ? robustLoss(2.0 * total_reprojection_loss) : total_reprojection_loss;
106  return loss;
107  } catch (...) {
108  // Catch cheirality and other errors, zero on errors
109  return 0.0;
110  }
111  } else { // Inactive
112  return 0.0;
113  }
114  }

146  // More efficient implementation of robust loss (also avoids inheritance calls)
147  double robustLoss(const double mahal_distance) const {
148  const double sqrt_mahal_distance = std::sqrt(mahal_distance);
149  const double absError = std::abs(sqrt_mahal_distance);
150  if (absError <= huber_k_) { // |x| <= k
151  return mahal_distance / 2;
152  } else { // |x| > k
153  return huber_k_ * (absError - (huber_k_ / 2));
154  }
155  }

157  bool valid(const Values& values) const {
158  typename Base::Cameras cameras = this->cameras(values);
159  const auto point = this->triangulateSafe(cameras);
160  if (point.valid()) {
161  return true;
162  } else if (useForRotationOnly(point)) {
163  Unit3 backProjected = cameras[0].backprojectPointAtInfinity(this->measured().at(0));
164  try {
165  cameras.reprojectionError(backProjected, this->measured());
166  } catch (...) {
167  return false;
168  }
169  return true;
170  } else {
171  return false;
172  }
173  // Shouldn't get here
174  return false;
175  }

177  private:
178  template <size_t D, size_t ZDim>
179  boost::shared_ptr<RegularJacobianFactor<D>> createRegularJacobianFactorSVD(
180  const KeyVector& keys,
181  const std::vector<Eigen::Matrix<double, ZDim, D>, Eigen::aligned_allocator<Eigen::Matrix<double, ZDim, D>>>&
182  Fblocks,
183  const Matrix& Enull, const Vector& b) const {
184  typedef std::pair<Key, Matrix> KeyMatrix;
185 
186  Vector reduced_error = Enull.transpose() * b;
187  // Apply noise whitening and robust weighting manually to more efficiently robustify
188  // error vector and jacobians. Equivalent to calling whitenSystem with a robust noise model.
189  // Assumes noise is diagonal (required for this factor anyway).
190  reduced_error *= noise_inv_sigma_;
191  double reduced_matrix_weight = noise_inv_sigma_;
192  if (robust_) {
193  const double robust_weight = robustWeight(reduced_error.norm());
194  reduced_error *= robust_weight;
195  reduced_matrix_weight *= robust_weight;
196  }
197 
198  size_t numKeys = Enull.rows() / ZDim;
199  size_t m2 = Enull.cols();
200  std::vector<KeyMatrix> reduced_matrices;
201  reduced_matrices.reserve(numKeys);
202  for (size_t k = 0; k < Fblocks.size(); ++k) {
203  Key key = keys[k];
204  reduced_matrices.emplace_back(
205  KeyMatrix(key, (Enull.transpose()).block(0, ZDim * k, m2, ZDim) * Fblocks[k] * reduced_matrix_weight));
206  }
207 
208  return boost::make_shared<RegularJacobianFactor<D>>(reduced_matrices, reduced_error);
209  }
210 
211  // TODO(rsoussan): profile these calls vs. gtsam better, is there a significant improvement?
212  // More efficient implementation of robust weight (also avoids inheritance calls)
213  double robustWeight(const double error_norm) const {
214  const double squared_weight = (error_norm <= huber_k_) ? (1.0) : (huber_k_ / error_norm);
215  return std::sqrt(squared_weight);
216  }
