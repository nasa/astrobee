429 template <typename T>
430 void toROSMsg(const pcl::PointCloud<T>& cloud, sensor_msgs::msg::Image& msg) {
431  // Ease the user's burden on specifying width/height for unorganized datasets
432  if (cloud.width == 0 && cloud.height == 0) {
433  throw std::runtime_error("Needs to be a dense like cloud!!");
434  } else {
435  if (cloud.points.size() != cloud.width * cloud.height)
436  throw std::runtime_error("The width and height do not match the cloud size!");
437  msg.height = cloud.height;
438  msg.width = cloud.width;
439  }
440 
441  // sensor_msgs::image_encodings::BGR8;
442  msg.encoding = "bgr8";
443  msg.step = msg.width * sizeof(std::uint8_t) * 3;
444  msg.data.resize(msg.step * msg.height);
445  for (size_t y = 0; y < cloud.height; y++) {
446  for (size_t x = 0; x < cloud.width; x++) {
447  std::uint8_t* pixel = &(msg.data[y * msg.step + x * 3]);
448  memcpy(pixel, &cloud(x, y).rgb, 3 * sizeof(std::uint8_t));
449  }
450  }
451 }

528  inline bool concatenatePointCloud(const sensor_msgs::msg::PointCloud2& cloud1,
529  const sensor_msgs::msg::PointCloud2& cloud2,
530  sensor_msgs::msg::PointCloud2& cloud_out) {
531  // if one input cloud has no points, but the other input does, just return the cloud with points
532  if (cloud1.width * cloud1.height == 0 && cloud2.width * cloud2.height > 0) {
533  cloud_out = cloud2;
534  return (true);
535  } else if (cloud1.width * cloud1.height > 0 && cloud2.width * cloud2.height == 0) {
536  cloud_out = cloud1;
537  return (true);
538  }
539 
540  bool strip = false;
541  for (size_t i = 0; i < cloud1.fields.size (); ++i)
542  if (cloud1.fields[i].name == "_")
543  strip = true;
544
