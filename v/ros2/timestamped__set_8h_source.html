149 template <typename T>
150 std::pair<boost::optional<TimestampedValue<T>>, boost::optional<TimestampedValue<T>>>
151 TimestampedSet<T>::LowerAndUpperBound(const Time timestamp) const {
152  if (empty()) {
153  LogDebug("LowerAndUpperBound: No timestamps available.");
154  return {boost::none, boost::none};
155  }
156 
157  // lower bound returns first it >= query, call this upper bound
158  const auto upper_bound_it = timestamp_value_map_.lower_bound(timestamp);
159  if (upper_bound_it == timestamp_value_map_.cend()) {
160  LogDebug("LowerAndUpperBoundTimestamps: No upper bound timestamp exists.");
161  return {boost::optional<TimestampedValue<T>>(*(timestamp_value_map_.crbegin())), boost::none};
162  } else if (upper_bound_it == timestamp_value_map_.cbegin()) {
163  LogDebug("LowerAndUpperBoundTimestamps: No lower bound timestamp exists.");
164  return {boost::none, boost::optional<TimestampedValue<T>>(*upper_bound_it)};
165  }
166  const auto lower_bound_it = std::prev(upper_bound_it);
167 
168  return {boost::optional<TimestampedValue<T>>(*lower_bound_it), boost::optional<TimestampedValue<T>>(*upper_bound_it)};
169 }

171 template <typename T>
172 boost::optional<TimestampedValue<T>> TimestampedSet<T>::LowerBoundOrEqual(const Time timestamp) const {
173  const auto lower_and_upper_bound_values = LowerAndUpperBound(timestamp);
174  if (!lower_and_upper_bound_values.first && !lower_and_upper_bound_values.second) {
175  LogDebug("LowerBoundOrEqual: Failed to get lower or upper bound values.");
176  return boost::none;
177  }
178 
179  // Only return upper bound values if it is equal to timestamp
180  if (lower_and_upper_bound_values.second && lower_and_upper_bound_values.second->timestamp == timestamp) {
181  return lower_and_upper_bound_values.second;
182  }
183 
184  return lower_and_upper_bound_values.first;
185 }

202 template <typename T>
203 boost::optional<TimestampedValue<T>> TimestampedSet<T>::Closest(const Time timestamp) const {
204  if (empty()) {
205  LogDebug("Closest: No values available.");
206  return boost::none;
207  }
208 
209  const auto lower_and_upper_bound_values = LowerAndUpperBound(timestamp);
210  if (!lower_and_upper_bound_values.first && !lower_and_upper_bound_values.second) {
211  LogDebug("Closest: Failed to get lower or upper bound values.");
212  return boost::none;
213  }
214 
215  if (!lower_and_upper_bound_values.first) {
216  return lower_and_upper_bound_values.second;
217  } else if (!lower_and_upper_bound_values.second) {
218  return lower_and_upper_bound_values.first;
219  } else {
220  const Time lower_bound_timestamp = lower_and_upper_bound_values.first->timestamp;
221  const Time upper_bound_timestamp = lower_and_upper_bound_values.second->timestamp;
222  const double upper_bound_dt = std::abs(timestamp - upper_bound_timestamp);
223  const double lower_bound_dt = std::abs(timestamp - lower_bound_timestamp);
224  return (upper_bound_dt < lower_bound_dt) ? lower_and_upper_bound_values.second : lower_and_upper_bound_values.first;
225  }
226 }

244 template <typename T>
245 int TimestampedSet<T>::RemoveOldValues(const Time oldest_allowed_timestamp) {
246  int num_removed_values = 0;
247  for (auto it = timestamp_value_map_.begin(); it != timestamp_value_map_.end();) {
248  if (it->first >= oldest_allowed_timestamp) break;
249  it = timestamp_value_map_.erase(it);
250  ++num_removed_values;
251  }
252 
253  return num_removed_values;
254 }
