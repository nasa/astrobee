33 // This is a simple wrapper around the ROS2 timer. The ROS2 timer API is a bit
34 // different than the ROS1 timer. This wrapper provides ROS1 timer functionality

105  void reset() {
106  if (created_) {
107  timer_->reset();
108  }
109  }

111  // The ROS1 timer has a reset parameter that was set to true by default.
112  // ARS never passed false for reset so the following two functions don't
113  // take a reset argument and implement the reset is true functionality
114  // from ROS1
115  bool setPeriod(double period) {
116  int period_sec = std::floor(period);
117  int period_nsec = (period - period_sec) * 1e9;
118  rclcpp::Duration d(period_sec, period_nsec);
119  return setPeriod(d);
120  }
121 
122  bool setPeriod(rclcpp::Duration period) {
123  if (period.toSec() < 0.0) period = rclcpp::Duration(0, 0);
124  period_ = period;
125  // See if the timer has been created
126  if (created_) {
127  // Check if the timer is stopped
128  if (timer_->is_canceled()) {
129  // Don't want to start the timer if it isn't running
130  created_ = false;
131  } else {
132  timer_->cancel();
133  return createStartTimer();
134  }
135  }
136  return true;
137  }
138 
139  // Follows the ros1 one convention that it doesn't do anything if the timer
140  // is already started
141  bool start() {
142  if (!created_) {
143  return createStartTimer();
144  } else if (timer_->is_canceled()) {
145  timer_->reset();
146  }
147  return true;
148  }
149 
150  void stop() {
151  if (created_) {
152  timer_->cancel();
153  }
154  }
155 
156  protected:
157  bool createStartTimer() {
158  if (node_ && clock_) {
159  timer_ = rclcpp::create_timer(node_,
160  clock_,
161  period_,
162  std::bind(&FreeFlyerTimer::TimerCallback, this));
163  created_ = true;
164  } else {
165  FF_DEFINE_LOGGER("ff_timer")
166  FF_ERROR("Node and clock are null when trying to create timer.");
167  created_ = false;
168  }
169  return created_;
170  }
171 
172  // Called when the timer times out
173  void TimerCallback() {
174  if (one_shot_) {
175  timer_->cancel();
176  }
177  if (cb_timer_) {
178  cb_timer_();
179  }
180  }
181 
182  protected:
183  bool one_shot_, created_, initialized_;
184  rclcpp::Duration period_;
185  TimerCallbackType cb_timer_;
186  rclcpp::Clock::SharedPtr clock_;
187  rclcpp::Node::SharedPtr node_;
188  rclcpp::TimerBase::SharedPtr timer_;
189 };
