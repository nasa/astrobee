40 // Possible result codes
41 class FreeFlyerActionState {
42  public:
43  enum Enum {
44  SUCCESS = 1, // Goal succeeded
45  PREEMPTED = 0, // Newer goal preempted previous goal
46  ABORTED = -1, // Goal was aborted by the server (cancelled goals don't call back!)
47  TIMEOUT_ON_CONNECT = -2, // The server did not accept the goal within CONNECT_TIME
48  TIMEOUT_ON_ACTIVE = -3, // The goal did not go active within ACTIVE TIME
49  TIMEOUT_ON_RESPONSE = -4, // No feedback or response was received within WATCHDOG TIME
50  TIMEOUT_ON_DEADLINE = -5, // No result received the given deadline
51  };
52 };

54 // Major changes from ros1 to ros2 are ros2 uses both topics and services to
55 // implement actions whereas ros1 only used topics. In ros1, new goals
56 // preempted/canceled the current goal. In ros2, an action server can accept
57 // multiple goals. This class attempts to apply the ros1 functionality to ros2.
58 // Please use the ros2 action server if you want to accept multiple goals.
59 template < typename ActionType >
60 class FreeFlyerActionServer {
61  private:
62  // For the purpose of this class, preempt will refer to the case of where
63  // a new goal was received before the original goal was accepted and
64  // canceled will refer to the case where a new goal was received
65  // before the original goal completed.
66  // In ros1 the server preempt callback was used for both preempting a goal
67  // and canceling a goal so the cancelled state in the above explanation will
68  // map to the preempt callback. The new preempt will be invisible to the code
69  // using this class.
70  enum State {
71  WAITING_FOR_CREATE,
72  WAITING_FOR_GOAL,
73  WAITING_FOR_ACCEPTED,
74  WAITING_FOR_RESPONSE,
75  PREEMPTED,
76  CANCELED
77  };
78 
79  public:
80  using GoalHandle = rclcpp_action::ServerGoalHandle<ActionType>;
81 
82  // Callback types
83  typedef std::function < void (std::shared_ptr<const typename ActionType::Goal>)> GoalCallbackType;
84  typedef std::function < void (void) > PreemptCallbackType;
85  typedef std::function < void (void) > CancelCallbackType;
86 
87  // Constructor
88  FreeFlyerActionServer() : result_sent_(false), state_(WAITING_FOR_CREATE) {}

91  ~FreeFlyerActionServer() {}

119  // Send incremental feedback for the current goal
120  void SendFeedback(std::shared_ptr<typename ActionType::Feedback> const
121  feedback) {
122  if (!current_goal_handle_ || !feedback) {
123  return;
124  }
125  current_goal_handle_->publish_feedback(feedback);
126  }

128  // Send the final result for the current goal
129  void SendResult(FreeFlyerActionState::Enum result_code,
130  std::shared_ptr<typename ActionType::Result> const result) {
131  if (!current_goal_handle_ || !result) {
132  return;
133  }
134 
135  switch (result_code) {
136  case FreeFlyerActionState::SUCCESS: // Everything worked
137  // In ros1, this is where canceled results would go since we were able
138  // to successful cancel. Ros2 has an actual canceled function. From the
139  // ros2 action diagram, it looks like it is fine to call succeed in the
140  // canceling state
141  current_goal_handle_->succeed(result);
142  break;
143  case FreeFlyerActionState::PREEMPTED:
144  case FreeFlyerActionState::ABORTED: // Server encounters an error
145  current_goal_handle_->abort(result);
146  break;
147  default:
148  break;
149  }
150 
151  result_sent_ = true;
152 
153  // Check to see if this goal was canceled and there is another goal waiting
154  // to be executed
155  if (state_ == CANCELED) {
156  if (current_goal_handle_->get_goal_id() != latest_uuid_) {
157  // Check if the accepted callback was called yet. If not, the goal will
158  // be started in the accepted callback
159  if (next_goal_handle_) {
160  // In this case, the goal was accepted and deferred in the goal
161  // callback. Need to set the goal to executing
162  next_goal_handle_->execute();
163  StartGoal(next_goal_handle_);
164  next_goal_handle_ = nullptr;
165  return;
166  } else {
167  current_goal_handle_ = nullptr;
168  // Stay in the canceled state so that execute will be called on the
169  // goal in the accepted callback
170  return;
171  }
172  }
173  }
174 
175  current_goal_handle_ = nullptr;
176  state_ = WAITING_FOR_GOAL;
177  }
178 
179  protected:
180  // A new action has been called. We have to care of a special case where the
181  // goal is preempted or cancelled.
182  rclcpp_action::GoalResponse GoalCallback(
183  const rclcpp_action::GoalUUID & uuid,
184  std::shared_ptr<const typename ActionType::Goal> goal) {
185  // ROS2 actions are a bit different in that you can reject a goal in the
186  // goal callback. So we don't want to start executing the goal in this
187  // callback. We will started executing it in the accepted callback.
188  // Also, ROS2 allows multiple goals to be executed at one time and this
189  // doesn't deal with preemption. Thus this wrapper class will implement
190  // preemption. Another note is ROS2 actions are built on topics and services
191  // so it is important that the goal and accepted callbacks return quickly to
192  // avoid blocking the executor
193 
194  // Not sure if this is correct but the goal is part of the goal handle in
195  // the accepted callback so the goal passed in to this callback is not
196  // being saved.
197 
198  latest_uuid_ = uuid;
199  // If we are waiting for last goal to go active, this new goal is preempting
200  // the last one. Not sure how likely it is to receive a third or forth goal
201  // before the first goal is accepted but let's handle it just in case
202  if (state_ == WAITING_FOR_ACCEPTED || state_ == PREEMPTED) {
203  state_ = PREEMPTED;
204  // Accept and start executing the goal in the accepted callback. We will
205  // kill previous goals in their accept callback
206  return rclcpp_action::GoalResponse::ACCEPT_AND_EXECUTE;
207  } else if (state_ == WAITING_FOR_RESPONSE) {
208  // If the current goal is being execution, we need to cancel it.
209  // In the ros1 astrobee code, this was referred to as preemption
210  state_ = CANCELED;
211  result_sent_ = false;
212  if (cb_preempt_) {
213  cb_preempt_();
214  }
215  // Check if the result was send. If it was, we can accept and execute the
216  // the goal. Otherwise, we need to accept but don't start executing the
217  // goal until the previous goal has been completed.
218  if (result_sent_) {
219  // Set state to waiting for accepted so we don't call execute on an
220  // already executing goal
221  state_ = WAITING_FOR_ACCEPTED;
222  return rclcpp_action::GoalResponse::ACCEPT_AND_EXECUTE;
223  } else {
224  return rclcpp_action::GoalResponse::ACCEPT_AND_DEFER;
225  }
226  } else if (state_ == CANCELED) {
227  // We are receive yet another goal before working on the last goal or a
228  // user canceled a goal and sent a new one before the cancel finished.
229  return rclcpp_action::GoalResponse::ACCEPT_AND_DEFER;
230  }
231 
232  // Nominal case where we are waiting for goal
233  state_ = WAITING_FOR_ACCEPTED;
234  return rclcpp_action::GoalResponse::ACCEPT_AND_EXECUTE;
235  }
236 
237  void AcceptedCallback(const std::shared_ptr<GoalHandle> goal_handle) {
238  // First check to see if this goal id matches the latest goal id. If it
239  // doesn't we don't care about this goal
240  if (goal_handle->get_goal_id() != latest_uuid_) {
241  std::shared_ptr<typename ActionType::Result> result =
242  std::make_shared<typename ActionType::Result>();
243  if (state_ == WAITING_FOR_ACCEPTED) {
244  // This cancel will be called when there is one goal that got canceled
245  // before its accepted callback was called. This should be very rare.
246  goal_handle->canceled(result);
247  } else {
248  // This abort will get called if multiple goals were received before
249  // the accepted callback was called. It is unlikely that more than one
250  // goal will preempt the original of canceled goal. However, this code
251  // should be able to handle that case. Also this isn't really an abort
252  // but the canceled function documentation makes it sound like the goal
253  // had to be canceled by the action client before it can be canceled in
254  // the server. Since these cases are very unlikely, output a warning in
255  // case debugging is needed.
256  FF_DEBUG("Rare abort goal case. This may need some attention.");
257  goal_handle->abort(result);
258  }
259  } else {
260  // Check if nominal or preempted case. For preempted, we already checked
261  // that this was the most recent goal handle
262  if (state_ == WAITING_FOR_ACCEPTED || state_ == PREEMPTED) {
263  StartGoal(goal_handle);
264  } else if (state_ == CANCELED) {
265  // First check to see if the old goal has been successfully canceled. We
266  // don't want to start the new goal until the old one was successfully
267  // canceled.
268  if (!current_goal_handle_) {
269  // In this case, the goal was accepted and deferred in the goal
270  // callback. Need to set the goal to executing
271  goal_handle->execute();
272  StartGoal(goal_handle);
273  } else {
274  // If the old goal still needs to be canceled, save the goal handle
275  next_goal_handle_ = goal_handle;
276  }
277  } else {
278  // Handle the case were we get a goal handle and don't know what to do
279  FF_ERROR("ff action: Got accepted goal callback in state %d.", state_);
280  std::shared_ptr<typename ActionType::Result> result =
281  std::make_shared<typename ActionType::Result>();
282  goal_handle->abort(result);
283  }
284  }
285  }
286 
287  void StartGoal(const std::shared_ptr<GoalHandle> goal_handle) {
288  state_ = WAITING_FOR_RESPONSE;
289  current_goal_handle_ = goal_handle;
290  if (cb_goal_) {
291  std::thread(cb_goal_, current_goal_handle_->get_goal()).detach();
292  }
293  }
294 
295  // Handles request to cancel a goal
296  rclcpp_action::CancelResponse CancelCallback(
297  const std::shared_ptr<GoalHandle> goal_handle) {
298  // Nominal case
299  if (state_ == WAITING_FOR_RESPONSE) {
300  FF_INFO("FFS: In waiting for response if.");
301  // A result must be sent when a goal is canceled. If not, the action
302  // server will throw an exception.
303  result_sent_ = false;
304  state_ = CANCELED;
305  if (cb_cancel_) {
306  cb_cancel_();
307  }
308  // Send result if it wasn't sent
309  if (!result_sent_) {
310  std::shared_ptr<typename ActionType::Result> result =
311  std::make_shared<typename ActionType::Result>();
312  // Weirdly sending a canceled result causes a failure/exception in the
313  // action server so send succeeded instead which totally makes sense
314  goal_handle->succeed(result);
315  }
316  } else if (state_ == WAITING_FOR_ACCEPTED) {
317  // This assumes actions allow a goal to be canceled before fully being
318  // accepted
319  // Make latest uuid invalid so we have a way to cancel the goal in the
320  // accepted callback
321  latest_uuid_ = {};
322  } else if (state_ == PREEMPTED) {
323  // If the cancel was not called on the latest goal, the goal will be
324  // aborted anyway. Do we care if it is aborted instead of cancelled?
325  if (latest_uuid_ == goal_handle->get_goal_id()) {
326  // Set state to waiting for goal. When the accepted callback for the
327  // latest goal is called, the uuid check will fail and the goal will be
328  // aborted. Again, do we care that the goal was aborted instead of
329  // cancelled?
330  state_ = WAITING_FOR_GOAL;
331  latest_uuid_ = {};
332  }
333  } else if (state_ == CANCELED) {
334  if (current_goal_handle_->get_goal_id() != goal_handle->get_goal_id()) {
335  // Check to make sure the latest goal is being canceled. If we had
336  // multiple goals come in while the oldest was being canceled, they will
337  // be aborted in the accepted callback. Again, do we care that the goals
338  // were aborted instead of canceled?
339  if (latest_uuid_ == goal_handle->get_goal_id()) {
340  // If accepted was called for the new goal, cancel it and set latest
341  // uuid to the current goal uuid if that goal hasn't finished
342  if (next_goal_handle_) {
343  std::shared_ptr<typename ActionType::Result> result =
344  std::make_shared<typename ActionType::Result>();
345  next_goal_handle_->canceled(result);
346  latest_uuid_ = current_goal_handle_->get_goal_id();
347  next_goal_handle_ = nullptr;
348  } else {
349  // If accepted wasn't called yet, set the latest uuid to be invalid
350  // so that it gets aborted in the accepted callback. Again, do we
351  // care that the goal gets aborted instead of canceled?
352  latest_uuid_ = {};
353  }
354  }
355  }
356  }
357  return rclcpp_action::CancelResponse::ACCEPT;
358  }
359 
360  protected:
361  bool result_sent_;
362  State state_;
363  rclcpp_action::GoalUUID latest_uuid_;
364  typename rclcpp_action::Server<ActionType>::SharedPtr sas_;
365  typename std::shared_ptr<GoalHandle> current_goal_handle_;
366  typename std::shared_ptr<GoalHandle> next_goal_handle_;
367  GoalCallbackType cb_goal_;
368  PreemptCallbackType cb_preempt_;
369  CancelCallbackType cb_cancel_;
370 };

392 template < typename ActionType >
393 class FreeFlyerActionClient {
394  private:
395  enum State {
396  WAITING_FOR_CREATE,
397  WAITING_FOR_CONNECT,
398  WAITING_FOR_GOAL,
399  WAITING_FOR_ACTIVE,
400  WAITING_FOR_RESPONSE,
401  WAITING_FOR_DEADLINE
402  };
403  static constexpr double DEFAULT_TIMEOUT_CONNECTED = 0.0;
404  static constexpr double DEFAULT_TIMEOUT_ACTIVE = 0.0;
405  static constexpr double DEFAULT_TIMEOUT_RESPONSE = 0.0;
406  static constexpr double DEFAULT_TIMEOUT_DEADLINE = 0.0;
407  static constexpr double DEFAULT_POLL_DURATION = 0.1;
408 
409  public:
410  using GoalHandle = rclcpp_action::ClientGoalHandle<ActionType>;
411 
412  // Callback types
413  typedef std::function < void (
414  std::shared_ptr<const typename ActionType::Feedback>) > FeedbackCallbackType;
415  typedef std::function < void (FreeFlyerActionState::Enum,
416  std::shared_ptr<const typename ActionType::Result>) > ResultCallbackType;
417  typedef std::function < void (void) > ConnectedCallbackType;
418  typedef std::function < void (void) > ActiveCallbackType;
419 
420  // Constructor
421  FreeFlyerActionClient() : state_(WAITING_FOR_CREATE),
422  to_connect_(DEFAULT_TIMEOUT_CONNECTED),
423  to_active_(DEFAULT_TIMEOUT_ACTIVE),
424  to_response_(DEFAULT_TIMEOUT_RESPONSE),
425  to_deadline_(DEFAULT_TIMEOUT_DEADLINE),
426  to_poll_(DEFAULT_POLL_DURATION) {}
427 
428  // Destructor
429  ~FreeFlyerActionClient() {}
430 
431  // Setters for callbacks
432  void SetFeedbackCallback(FeedbackCallbackType cb_feedback) {
433  cb_feedback_ = cb_feedback;
434  }
435  void SetResultCallback(ResultCallbackType cb_result) {
436  cb_result_ = cb_result;
437  }
438  void SetConnectedCallback(ConnectedCallbackType cb_connected) {
439  cb_connected_ = cb_connected;
440  }
441  void SetActiveCallback(ActiveCallbackType cb_active) {
442  cb_active_ = cb_active;
443  }
444 
445  // Setters for timeouts
446  void SetConnectedTimeout(double to_connect) { to_connect_ = to_connect; }
447  void SetActiveTimeout(double to_active) { to_active_ = to_active; }
448  void SetResponseTimeout(double to_response) { to_response_ = to_response; }
449  void SetDeadlineTimeout(double to_deadline) { to_deadline_ = to_deadline; }
450  void SetPollTime(double to_poll) { to_poll_ = to_poll; }
451 
452  // Initialize the action client and return whether connected by end of call
453  bool Create(NodeHandle nh, std::string const& topic) {
454  // Initialize all timers, but do not start them.
455  timer_connected_.createTimer(to_connect_,
456  std::bind(&FreeFlyerActionClient::ConnectedTimeoutCallback, this),
457  nh, true, false);
458  timer_active_.createTimer(to_active_,
459  std::bind(&FreeFlyerActionClient::ActiveTimeoutCallback, this),
460  nh, true, false);
461  timer_response_.createTimer(to_response_,
462  std::bind(&FreeFlyerActionClient::ResponseTimeoutCallback, this),
463  nh, true, false);
464  timer_deadline_.createTimer(to_deadline_,
465  std::bind(&FreeFlyerActionClient::DeadlineTimeoutCallback, this),
466  nh, true, false);
467  timer_poll_.createTimer(to_poll_,
468  std::bind(&FreeFlyerActionClient::ConnectPollCallback, this),
469  nh, false, false);
470  // Initialize the action client
471  sac_ = rclcpp_action::create_client<ActionType>(nh, topic);
472  // Set the state
473  state_ = WAITING_FOR_GOAL;
474  // If we have a non-zero connect timeout, this means that we'd like to have
475  // the system poll for server ready in the background and callback
476  // when the server is ready
477  if (to_connect_ != 0) {
478  state_ = WAITING_FOR_CONNECT;
479  StartOptionalTimer(timer_connected_, to_connect_);
480  StartOptionalTimer(timer_poll_, to_poll_);
481  ConnectPollCallback();
482  }
483  // Return whether the server is ready by the end of this function
484  return IsConnected();
485  }
486 
487  // Check if connected
488  bool IsConnected() {
489  return (sac_ && sac_->action_server_is_ready());
490  }
491 
492  // Send a new goal
493  bool SendGoal(typename ActionType::Goal const& goal) {
494  if (!sac_) return false;
495 
496  // If we are preempting another goal, stop all timers.
497  if (current_goal_handle_) {
498  StopAllTimers();
499  }
500  // In ros1, if a previous goal is active and send goal is called, the old
501  // goal is simply forgotten and the new goal is tracked. To preserve this
502  // functionality, set the current goal_handle to null so we ignore any
503  // feedback or results received from the old goal. The current goal handle
504  // will then be assigned to the new goal once the goal is accepted and the
505  // accepted callback is called.
506  current_goal_handle_ = nullptr;
507 
508  // Start the active timer waiting for goal to be accepted
509  StartOptionalTimer(timer_active_, to_active_);
510  StartOptionalTimer(timer_deadline_, to_deadline_);
511  typename rclcpp_action::Client<ActionType>::SendGoalOptions sgo =
512  typename rclcpp_action::Client<ActionType>::SendGoalOptions();
513  sgo.goal_response_callback =
514  std::bind(&FreeFlyerActionClient::ActiveCallback,
515  this,
516  std::placeholders::_1);
517  sgo.feedback_callback =
518  std::bind(&FreeFlyerActionClient::FeedbackCallback,
519  this,
520  std::placeholders::_1,
521  std::placeholders::_2);
522  sgo.result_callback = std::bind(&FreeFlyerActionClient::ResultCallback,
523  this,
524  std::placeholders::_1);
525  // Send the goal
526  sac_->async_send_goal(goal, sgo);
527  // Update the state
528  state_ = WAITING_FOR_ACTIVE;
529  return true;
530  }
531 
532  // Cancel the goal that is currently running
533  bool CancelGoal() {
534  if (!sac_ || !current_goal_handle_) return false;
535  // Only cancel a goal if we are in the correct state
536  switch (state_) {
537  case WAITING_FOR_ACTIVE:
538  case WAITING_FOR_RESPONSE:
539  case WAITING_FOR_DEADLINE:
540  StopAllTimers();
541  sac_->async_cancel_goal(current_goal_handle_);
542  // Set goal handle to null so we stop tracking the goal
543  current_goal_handle_ = nullptr;
544  state_ = WAITING_FOR_GOAL;
545  return true;
546  default:
547  break;
548  }
549  return false;
550  }
551 
552  protected:
553  // Simple wrapper around an optional timer
554  void StartOptionalTimer(ff_util::FreeFlyerTimer & timer, double duration) {
555  if (duration == 0) return;
556  timer.stop();
557  timer.setPeriod(duration);
558  timer.start();
559  }

561  // Stop all timers
562  void StopAllTimers() {
563  timer_connected_.stop();
564  timer_active_.stop();
565  timer_response_.stop();
566  timer_deadline_.stop();
567  }
568 
569  // Completes the current goal
570  void Complete(FreeFlyerActionState::Enum state,
571  std::shared_ptr<const typename ActionType::Result> result) {
572  // Stop all timers
573  StopAllTimers();
574  // Send response
575  if (cb_result_)
576  cb_result_(state, result);
577  // Reset state
578  state_ = WAITING_FOR_GOAL;
579  current_goal_handle_ = nullptr;
580  }
581 
582  // Called periodically until the server is connected
583  void ConnectPollCallback() {
584  if (!sac_ || !sac_->action_server_is_ready()) return;
585  timer_connected_.stop();
586  timer_poll_.stop();
587  state_ = WAITING_FOR_GOAL;
588  if (cb_connected_)
589  cb_connected_();
590  }
591 
592  // Called when the server cannot be connected to
593  void ConnectedTimeoutCallback() {
594  Complete(FreeFlyerActionState::TIMEOUT_ON_CONNECT, nullptr);
595  }
596 
597  // Called when the goal does not go active
598  void ActiveTimeoutCallback() {
599  FF_WARN("Freeflyer action timed out on going active.");
600  CancelGoal();
601  Complete(FreeFlyerActionState::TIMEOUT_ON_ACTIVE, nullptr);
602  }
603 
604  // Called when the task deadline was not met
605  void DeadlineTimeoutCallback() {
606  CancelGoal();
607  Complete(FreeFlyerActionState::TIMEOUT_ON_DEADLINE, nullptr);
608  }
609 
610  // Called when no feedback/result is received within a certain period of time
611  void ResponseTimeoutCallback() {
612  CancelGoal();
613  Complete(FreeFlyerActionState::TIMEOUT_ON_RESPONSE, nullptr);
614  }
615 
616  // Goal is now active, restart timer and switch state
617  void ActiveCallback(typename GoalHandle::SharedPtr goal_handle) {
618  timer_active_.stop();
619 
620  if (!sac_) return;
621 
622  // In ros2, it is possible for a goal to be rejected. Check that here and
623  // save goal handle for later use.
624  current_goal_handle_ = goal_handle;
625  if (!current_goal_handle_) {
626  Complete(FreeFlyerActionState::ABORTED, nullptr);
627  } else {
628  // Start a responser timer if required
629  StartOptionalTimer(timer_response_, to_response_);
630  // If we want to know when goal goes active
631  if (cb_active_)
632  cb_active_();
633  // We are now waiting on a response
634  state_ = WAITING_FOR_RESPONSE;
635  }
636  }
637 
638  // Feedback received
639  void FeedbackCallback(typename GoalHandle::SharedPtr goal_handle,
640  const std::shared_ptr<const typename ActionType::Feedback> feedback) {
641  timer_response_.stop();
642  // Need to check if this is the newest goal being executed. We can ignore
643  // the feedback from preempted/canceled goals
644  if (current_goal_handle_ &&

651  }

654  // Called when a result is received
655  void ResultCallback(typename GoalHandle::WrappedResult const& result) {
656  if (current_goal_handle_ &&
657  current_goal_handle_->get_goal_id() == result.goal_id) {
658  // The response we send depends on the state
659  result_ = result.result;
660 
661  // The response we send depends on the state
662  // We don't track canceled goals so we don't need to worry about that
663  // return code
664  if (result.code == rclcpp_action::ResultCode::SUCCEEDED) {
665  state_response_ = FreeFlyerActionState::SUCCESS;
666  } else {
667  // Unfortunately the free flyer action server code had to abort
668  // preempted goals so these goals could have either been preempted or
669  // aborted. After some poking around in the free flyer code, it looks
670  // like the code only cares if the goal succeeded or failed. I.e. there
671  // is no check to distinguish between preempted or aborted. If the
672  // distinction is necessary, one can use the result since the response
673  // portion of the result in all free flyer actions is 0 for preemption.
674  state_response_ = FreeFlyerActionState::ABORTED;
675  }
676 
677  // Stop timers now. If another timer times out before the result delay
678  // timer, cancel goal will be called which in this case will try to cancel
679  // an already completed goal which will result in a very cryptic error
680  StopAllTimers();
681 
682  // Call the result callback on the client side
683  Complete(state_response_, result_);
684 
685  // Return to waiting for a goal
686  state_ = WAITING_FOR_GOAL;
687  }
688  }
