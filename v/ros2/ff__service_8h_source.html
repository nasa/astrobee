45 // This is a simple wrapper around a ROS2 service client.
46 
47 template < class ServiceSpec >
48 class FreeFlyerServiceClient {
49  protected:
50  enum State {
51  WAITING_FOR_CREATE = 0, // Connect() has not been called
52  WAITING_FOR_CONNECT = 1, // Waiting to connect to server
53  WAITING_FOR_CALL = 2 // Connection established, waiting on call
54  };
55  static constexpr double DEFAULT_TIMEOUT_CONNECTED = 10.0;
56  static constexpr double DEFAULT_POLL_DURATION = 0.1;
57 
58  public:
59  // Callback types
60  typedef std::function < void (void) > ConnectedCallbackType;
61  typedef std::function < void (void) > TimeoutCallbackType;
62 
63  // Setters for callbacks
64  void SetTimeoutCallback(TimeoutCallbackType cb_timeout) { cb_timeout_ = cb_timeout; }
65  void SetConnectedCallback(ConnectedCallbackType cb_connected) { cb_connected_ = cb_connected; }
66 
67  // Setters for timeouts
68  void SetConnectedTimeout(double to_connected) { to_connected_ = to_connected; }
69 
70  // Constructor
71  FreeFlyerServiceClient() : state_(WAITING_FOR_CREATE),
72  to_connected_(DEFAULT_TIMEOUT_CONNECTED),
73  to_poll_(DEFAULT_POLL_DURATION) {}
74 
75  // Destructor
76  ~FreeFlyerServiceClient() {}
77 
78  bool Create(rclcpp::Node::SharedPtr node, std::string const& topic) {
79  // Create a timer to poll to see if the server is ready
80  timer_connected_.createTimer(to_connected_,
81  std::bind(&FreeFlyerServiceClient::TimeoutCallback, this),
82  node,
83  true,
84  false);
85  timer_poll_.createTimer(to_poll_,
86  std::bind(&FreeFlyerServiceClient::ConnectPollCallback, this),
87  node,
88  false,
89  false);
90  // Save the node handle and topic to support reconnects
91  node_ = node;
92  topic_ = topic;
93  client_group_ = node->create_callback_group(rclcpp::CallbackGroupType::MutuallyExclusive);
94 
95  return IsConnected();
96  }
97 
98  // Check that the service is ready
99  bool IsConnected() {
100  if (exists())
101  return true;
102  ConnectPollCallback();
103  return false;
104  }
105 
106  bool Call(FreeFlyerService<ServiceSpec> & service) {
107  return call(service);
108  }
109 
110  bool Call(const typename ServiceSpec::Request & request, std::shared_ptr<typename ServiceSpec::Response> & response) {
111  return call(request, response);
112  }
113 
114  // ROS1 functions
115  bool call(const typename ServiceSpec::Request & request, std::shared_ptr<typename ServiceSpec::Response> & response) {
116  return call(std::make_shared<typename ServiceSpec::Request>(request), response);
117  }
118  bool call(const std::shared_ptr<typename ServiceSpec::Request> & request,
119  std::shared_ptr<typename ServiceSpec::Response> & response) {
120  if (IsConnected()) {
121  auto result = service_client_->async_send_request(request);
122  std::future_status status = result.wait_for((std::chrono::seconds)10); // timeout to guarantee a graceful finish
123 
124  if (status == std::future_status::ready) {
125  response = result.get();
126  return true;
127  }
128  }
129  return false;
130  }
131 
132  bool call(FreeFlyerService<ServiceSpec> & service) {
133  return call(service.request.get(), service.response);
134  }
135 
136  bool exists() {
137  if (service_client_.get() != NULL) {
138  return service_client_->service_is_ready();
139  }
140  return false;
141  }
142 
143  bool isValid() {
144  return exists();
145  }
146 
147  bool waitForExistence(std::shared_ptr<rclcpp::Duration> duration) {
148  return waitForExistence(duration->seconds());
149  }
150 
151  bool waitForExistence(double duration_sec) {
152  if (service_client_.get() != NULL) {
153  return service_client_->wait_for_service((std::chrono::duration<double>) duration_sec);
154  }
155  return false;
156  }
157 
158  protected:
159  // Simple wrapper around an optional timer
160  void StartOptionalTimer(ff_util::FreeFlyerTimer & timer,
161  double duration) {
162  if (duration == 0) return;
163  timer.stop();
164  timer.setPeriod(duration);
165  timer.start();
166  }
167 
168  // Called periodically until the server is ready
169  void ConnectPollCallback() {
170  // Case: connected
171  if (service_client_.get() != NULL &&
172  service_client_->service_is_ready()) {
173  if (state_ != WAITING_FOR_CALL) {
174  state_ = WAITING_FOR_CALL;
175  timer_connected_.stop();
176  timer_poll_.stop();
177  if (cb_connected_)
178  cb_connected_();
179  }
180  // Case: disconnected
181  } else {
182  service_client_ = node_->create_client<ServiceSpec>(topic_, rmw_qos_profile_services_default,
183  client_group_);
184  state_ = WAITING_FOR_CONNECT;
185  StartOptionalTimer(timer_connected_, to_connected_);
186  StartOptionalTimer(timer_poll_, to_poll_);
187  }
188  }
189 
190  // Called when the service doesn't go active
191  void TimeoutCallback() {
192  timer_connected_.stop();
193  timer_poll_.stop();
194  if (cb_timeout_)
195  cb_timeout_();
196  }
197 
198  protected :
199  State state_;
200  double to_connected_;
201  double to_poll_;
202  TimeoutCallbackType cb_timeout_;
203  ConnectedCallbackType cb_connected_;
204  ServiceClient<ServiceSpec> service_client_;
205  rclcpp::CallbackGroup::SharedPtr client_group_;
206  ff_util::FreeFlyerTimer timer_connected_;
207  ff_util::FreeFlyerTimer timer_poll_;
208  std::string topic_;
209  rclcpp::Node::SharedPtr node_;
210 };
