28 template <typename MessageType>
29 class MessageBuffer {
30  public:
31  explicit MessageBuffer(const MessageBufferParams& params) : params_(params) {}
32  // Assumes messages are buffered in time order
33  void BufferMessage(const MessageType& msg) {
34  const localization_common::Time timestamp = localization_common::TimeFromHeader(msg.header);
35  if (last_measurement_time_ && std::abs(*last_measurement_time_ - timestamp) < params_.min_msg_spacing) {
36  LOG(WARNING) << "BufferMessage: Dropping message that arrived too close to previous message.";
37  return;
38  }
39  msg_buffer_.emplace(timestamp, msg);
40  last_measurement_time_ = timestamp;
41  }
42 
43  boost::optional<MessageType> GetMessage(const localization_common::Time current_time) {
44  if (msg_buffer_.empty()) return boost::none;
45  if (current_time - msg_buffer_.cbegin()->first < params_.msg_delay) {
46  VLOG(2) << "GetMessage: Current time too close to message time, no message available.";
47  return boost::none;
48  }
49  const auto msg = msg_buffer_.cbegin()->second;
50  msg_buffer_.erase(msg_buffer_.begin());
51  return msg;
52  }
53 
54  private:
55  MessageBufferParams params_;
56  std::map<localization_common::Time, MessageType> msg_buffer_;
57  boost::optional<localization_common::Time> last_measurement_time_;
58 };
