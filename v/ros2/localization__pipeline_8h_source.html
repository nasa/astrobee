48 enum PipelineError : uint8_t {
49  ERROR_REGISTRATION_TIMEOUT = 0, // Registration timeout
50  ERROR_VISUAL_TIMEOUT = 1, // Visual feature timeout
51  ERROR_DEPTH_TIMEOUT = 2, // Depth timeout
52  ERROR_FILTER_TIMEOUT = 3, // Filter timeout
53  ERROR_FILTER_BIAS = 4 // Filter is sampling the bias
54 };
55 
56 typedef std::function<void(PipelineError)> PipelineCallbackType;

58 struct Pipeline {
59  enum Components : uint8_t {
60  COMPONENT_FILTER = (1<<0), // Filter needed
61  COMPONENT_OPTICAL_FLOW = (1<<1), // Optical flow needed
62  COMPONENT_ENABLE = (1<<2), // Needs optical flow
63  COMPONENT_SET_INPUT = (1<<3), // Needs EKF input to be set
64  COMPONENT_REGISTRATIONS = (1<<4), // Needs registration pulses
65  COMPONENT_VISUAL_FEATURES = (1<<5), // Needs viusal features
66  COMPONENT_DEPTH_FEATURES = (1<<6) // Needs depth features
67  };
68 
69  // Constructor
70  Pipeline(uint8_t mode, std::string const& name)
71  : mode_(mode), components_(0), name_(name) {}
72 
73  // If the pipeline requires
74  Pipeline& NeedsFilter(uint32_t max_conf, bool optical_flow, double timeout) {
75  components_ |= COMPONENT_FILTER;
76  max_filter_ = max_conf;
77  if (optical_flow)
78  components_ |= COMPONENT_OPTICAL_FLOW;
79  timeout_filter_ = timeout;
80  return *this;
81  }
82 
83  // Needs to be enabled/disabled
84  Pipeline& NeedsEnable(std::string const& topic, double timeout) {
85  components_ |= COMPONENT_ENABLE;
86  topic_enable_ = topic;
87  timeout_enable_ = timeout;
88  return *this;
89  }
90 
91  // Needs measurements to be streamed
92  Pipeline& NeedsRegistrations(std::string const& topic, double timeout) {
93  components_ |= COMPONENT_REGISTRATIONS;
94  topic_reg_ = topic;
95  timeout_reg_ = timeout;
96  return *this;
97  }
98 
99  // Needs features
100  Pipeline& NeedsVisualFeatures(
101  std::string const& topic, double timeout, uint32_t min_features) {
102  components_ |= COMPONENT_VISUAL_FEATURES;
103  topic_visual_ = topic;
104  timeout_visual_ = timeout;
105  min_visual_ = min_features;
106  return *this;
107  }
108 
109  // Needs features
110  Pipeline& NeedsDepthFeatures(
111  std::string const& topic, double timeout, uint32_t min_features) {
112  components_ |= COMPONENT_DEPTH_FEATURES;
113  topic_depth_ = topic;
114  timeout_depth_ = timeout;
115  min_depth_ = min_features;
116  return *this;
117  }
118 
119  // Get the name of the pipeline
120  std::string const& GetTopic() {
121  return topic_enable_;
122  }
123 
124  // Get the name of the pipeline
125  std::string const& GetName() {
126  return name_;
127  }
128 
129  // Get the name of the pipeline
130  uint8_t GetMode() {
131  return mode_;
132  }
133 
134  // Get whether the pipeline needs optical flow
135  bool RequiresOpticalFlow() {
136  return (components_ & COMPONENT_OPTICAL_FLOW);
137  }
138 
139  // Get whether the pipeline needs optical flow
140  bool RequiresFilter() {
141  return (components_ & COMPONENT_FILTER);
142  }
143 
144  // Initialize the pipeline
145  bool Initialize(rclcpp::Node::SharedPtr nh,
146  PipelineCallbackType cb_error,
147  std::function<void(void)> cb_connected,
148  std::function<void(void)> cb_timeout) {
149  // Copy the callback and node handle
150  callback_ = cb_error;
151  nh_ = nh;
152  // If we need to enable
153  if (components_ & COMPONENT_ENABLE) {
154  service_.SetConnectedTimeout(timeout_enable_);
155  service_.SetConnectedCallback(cb_connected);
156  service_.SetTimeoutCallback(cb_timeout);
157  service_.Create(nh, topic_enable_);
158  }
159  // Disable by default
160  return true;
161  }
162 
163  // Check if the pipeline is initialized / connected to its services
164  bool IsConnected() {
165  if ((components_ & COMPONENT_ENABLE) && !service_.IsConnected())
166  return false;
167  return true;
168  }
169 
170  // Enable or disable this pipeline
171  bool Toggle(bool enable) {
172  // If we are enabling this pipeline
173  if (enable) {
174  if (components_ & COMPONENT_REGISTRATIONS) {
175  sub_reg_ = FF_CREATE_SUBSCRIBER(nh_, ff_msgs::CameraRegistration, topic_reg_, 1,
176  std::bind(&Pipeline::RegistrationCallback, this, std::placeholders::_1));
177  timer_reg_.createTimer(timeout_reg_,
178  std::bind(&Pipeline::RegistrationTimeoutCallback, this), nh_);
179  }
180  if (components_ & COMPONENT_VISUAL_FEATURES) {
181  sub_visual_ = FF_CREATE_SUBSCRIBER(nh_, ff_msgs::VisualLandmarks, topic_visual_, 1,
182  std::bind(&Pipeline::VisualCallback, this, std::placeholders::_1));
183  timer_visual_.createTimer(timeout_visual_,
184  std::bind(&Pipeline::VisualTimeoutCallback, this), nh_, false, false);
185  }
186  if (components_ & COMPONENT_DEPTH_FEATURES) {
187  sub_depth_ = FF_CREATE_SUBSCRIBER(nh_, ff_msgs::DepthLandmarks, topic_depth_, 1,
188  std::bind(&Pipeline::DepthCallback, this, std::placeholders::_1));
189  timer_depth_.createTimer(timeout_depth_,
190  std::bind(&Pipeline::DepthTimeoutCallback, this), nh_);
191  }
192  } else {
193  // Make sure we don't keep monitoring the filter when the pipeline
194  // is shutdown.
195  Use(false);
196  // Stop monitoring the raw pipeline stream.
197  if (components_ & COMPONENT_VISUAL_FEATURES) {
198  sub_visual_.reset();
199  timer_visual_.stop();
200  }
201  if (components_ & COMPONENT_DEPTH_FEATURES) {
202  sub_depth_.reset();
203  timer_depth_.stop();
204  }
205  if (components_ & COMPONENT_REGISTRATIONS) {
206  sub_reg_.reset();
207  timer_reg_.stop();
208  }
209  }
210  // Enable or disable the pipeline if needed
211  if (components_ & COMPONENT_ENABLE) {
212  ff_msgs::SetBool::Request req;
213  req.enable = enable;
214  auto res = std::make_shared<ff_msgs::SetBool::Response>();
215  if (!service_.Call(req, res))
216  return false;
217  }
218  return true;
219  }
220 
221  // Start actually using the pipeline
222  bool Use(bool enable) {
223  // If we are enabling this pipeline
224  if (enable) {
225  if (components_ & COMPONENT_FILTER) {
226  sub_filter_ = FF_CREATE_SUBSCRIBER(nh_, ff_msgs::EkfState, TOPIC_GNC_EKF, 1,
227  std::bind(&Pipeline::FilterCallback, this, std::placeholders::_1));
228  timer_filter_.createTimer(timeout_filter_,
229  std::bind(&Pipeline::FilterTimeoutCallback, this), nh_, false, false);
230  }
231  } else {
232  if (components_ & COMPONENT_FILTER) {
233  sub_filter_.reset();
234  timer_filter_.stop();
235  }
236  }
237  return true;
238  }
239 
240  protected:
241  // Called back when registration pulses arrive for processing
242  void RegistrationCallback(const std::shared_ptr<ff_msgs::CameraRegistration> msg) {
243  timer_reg_.stop();
244  timer_reg_.start();
245  }
246 
247  // Called when registration pulses don't arrive
248  void RegistrationTimeoutCallback() {
249  callback_(ERROR_REGISTRATION_TIMEOUT);
250  }
251 
252  // Called back when visual features arrive for processing
253  void VisualCallback(const std::shared_ptr<ff_msgs::VisualLandmarks> msg) {
254  if (msg->landmarks.size() < min_visual_)
255  return;
256  timer_visual_.stop();
257  timer_visual_.start();
258  }
259 
260  // Called when visual features don't arrive
261  void VisualTimeoutCallback() {
262  callback_(ERROR_VISUAL_TIMEOUT);
263  }
264 
265  // Called back when depth features arrive for processing
266  void DepthCallback(const std::shared_ptr<ff_msgs::DepthLandmarks> msg) {
267  if (msg->landmarks.size() < min_depth_)
268  return;
269  timer_depth_.stop();
270  timer_depth_.start();
271  }
272 
273  // Called when depth features don't arrive
274  void DepthTimeoutCallback() {
275  callback_(ERROR_DEPTH_TIMEOUT);
276  }
277 
278  // Called when filter estimates arrive
279  void FilterCallback(const std::shared_ptr<ff_msgs::EkfState> msg) {
280  if (msg->estimating_bias)
281  callback_(ERROR_FILTER_BIAS);
282  else if (msg->confidence > max_filter_)
283  return;
284  timer_filter_.stop();
285  timer_filter_.start();
286  }
287 
288  // Called when filter estimates don't arrive
289  void FilterTimeoutCallback() {
290  callback_(ERROR_FILTER_TIMEOUT);
291  }
292 
293  private:
294  uint8_t mode_, components_;
295  std::string name_, topic_enable_, topic_reg_, topic_visual_, topic_depth_;
296  double timeout_reg_, timeout_visual_, timeout_depth_, timeout_enable_, timeout_filter_;
297  rclcpp::Subscription<ff_msgs::VisualLandmarks>::SharedPtr sub_visual_;
298  rclcpp::Subscription<ff_msgs::DepthLandmarks>::SharedPtr sub_depth_;
299  rclcpp::Subscription<ff_msgs::CameraRegistration>::SharedPtr sub_reg_;
300  rclcpp::Subscription<ff_msgs::EkfState>::SharedPtr sub_filter_;
301  uint32_t min_visual_, min_depth_, max_filter_;
302  ff_util::FreeFlyerServiceClient<ff_msgs::SetBool> service_;
303  rclcpp::Node::SharedPtr nh_;
304  ff_util::FreeFlyerTimer timer_visual_, timer_filter_, timer_depth_, timer_reg_;
305  PipelineCallbackType callback_;
306 };
