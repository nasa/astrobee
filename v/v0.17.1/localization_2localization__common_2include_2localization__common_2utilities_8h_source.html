196 template <int CostDim, int StateDim>
197 boost::optional<Eigen::Matrix<double, StateDim, StateDim>> LeastSquaresCovariance(
198  const std::vector<Eigen::Matrix<double, CostDim, StateDim>>& cost_jacobians) {
199  const int num_costs = cost_jacobians.size();
200  Eigen::MatrixXd stacked_jacobians(num_costs * CostDim, StateDim);
201  int row_index = 0;
202  for (const auto& jacobian : cost_jacobians) {
203  stacked_jacobians.block(row_index, 0, CostDim, StateDim) = jacobian;
204  row_index += CostDim;
205  }
206  const Eigen::Matrix<double, 6, 6> hessian = stacked_jacobians.transpose() * stacked_jacobians;
207  const Eigen::FullPivLU<Eigen::Matrix<double, StateDim, StateDim>> lu(hessian);
208  if (!lu.isInvertible()) return boost::none;
209  // Use lu inverse rather than hessian.inverse() for improved numerical stability
210  const Eigen::Matrix<double, 6, 6> covariance = lu.inverse();
211  return covariance;
212 }

214 template <typename T>
215 std::vector<T> Transform(const std::vector<T>& a_T_elements, const Eigen::Isometry3d& b_T_a) {
216  std::vector<T> b_T_elements;
217  for (const auto& a_T_element : a_T_elements) {
218  b_T_elements.emplace_back(b_T_a * a_T_element);
219  }
220  return b_T_elements;
221 }
