53 template <typename T>
54 Eigen::Matrix<T, 2, 1> RelativeCoordinates(const Eigen::Matrix<T, 2, 1>& absolute_point,
55  const Eigen::Matrix<T, 3, 3>& intrinsics) {
56  const T& f_x = intrinsics(0, 0);
57  const T& f_y = intrinsics(1, 1);
58  const T& p_x = intrinsics(0, 2);
59  const T& p_y = intrinsics(1, 2);
60 
61  const T& x = absolute_point[0];
62  const T& y = absolute_point[1];
63  const T relative_x = (x - p_x) / f_x;
64  const T relative_y = (y - p_y) / f_y;
65  return Eigen::Matrix<T, 2, 1>(relative_x, relative_y);
66 }
67 
68 template <typename T>
69 Eigen::Matrix<T, 2, 1> AbsoluteCoordinates(const Eigen::Matrix<T, 2, 1>& relative_point,
70  const Eigen::Matrix<T, 3, 3>& intrinsics) {
71  const T& f_x = intrinsics(0, 0);
72  const T& f_y = intrinsics(1, 1);
73  const T& p_x = intrinsics(0, 2);
74  const T& p_y = intrinsics(1, 2);
75  return Eigen::Matrix<T, 2, 1>(relative_point[0] * f_x + p_x, relative_point[1] * f_y + p_y);
76 }
