165 template <typename FactorType>
166 const std::vector<boost::shared_ptr<const FactorType>> GraphOptimizer::Factors() const {
167  typename std::vector<boost::shared_ptr<const FactorType>> factors;
168  for (const auto& factor : graph_factors()) {
169  const auto casted_factor = boost::dynamic_pointer_cast<const FactorType>(factor);
170  if (casted_factor) factors.emplace_back(casted_factor);
171  }
172  return factors;
173 }
