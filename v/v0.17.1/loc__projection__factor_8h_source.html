147  // TODO(rsoussan): Replace PinholeCamera with PinholePose?
148  Vector evaluateError(const Pose3& pose, boost::optional<Matrix&> H1 = boost::none) const {
149  try {
150  if (body_P_sensor_) {
151  if (H1) {
152  Matrix H0;
153  PinholeCamera<CALIBRATION> camera(pose.compose(*body_P_sensor_, H0), *K_);
154  Point2 reprojectionError(camera.project(landmark_point_, H1, boost::none, boost::none) - measured_);
155  *H1 = *H1 * H0;
156  return reprojectionError;
157  } else {
158  PinholeCamera<CALIBRATION> camera(pose.compose(*body_P_sensor_), *K_);
159  return camera.project(landmark_point_, H1, boost::none, boost::none) - measured_;
160  }
161  } else {
162  PinholeCamera<CALIBRATION> camera(pose, *K_);
163  return camera.project(landmark_point_, H1, boost::none, boost::none) - measured_;
164  }
165  } catch (CheiralityException& e) {
166  if (H1) *H1 = Matrix::Zero(2, 6);
167  if (verboseCheirality_)
168  std::cout << e.what() << ": Landmark moved behind camera " << DefaultKeyFormatter(this->key()) << std::endl;
169  if (throwCheirality_) throw CheiralityException(this->key());
170  }
171  return Vector2::Constant(0.0);
172  }
173 
174  bool cheiralityError(const Pose3& pose) const {
175  try {
176  if (body_P_sensor_) {
177  PinholeCamera<CALIBRATION> camera(pose.compose(*body_P_sensor_), *K_);
178  camera.project(landmark_point_) - measured_;
179  } else {
180  PinholeCamera<CALIBRATION> camera(pose, *K_);
181  camera.project(landmark_point_) - measured_;
182  }
183  } catch (CheiralityException& e) {
184  return true;
185  }
186  return false;
187  }
